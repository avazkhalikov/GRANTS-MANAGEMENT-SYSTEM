CREATE VIEW [dbo].[vProjectsN]
AS
SELECT     p.ProjectID, pi.TitleE, pi.TitleR, g.Name, g.NameRu, psl.ProposalStatusText, prji.AwardedAmt
FROM         dbo.Projects p LEFT OUTER JOIN
                      dbo.ProposalStatus ps ON p.ProjectID = ps.ProjectID INNER JOIN
                      dbo.ProposalStatusList psl ON psl.ProposalStatusID = ps.PropStatusID LEFT OUTER JOIN
                      dbo.ProposalInfo pi ON pi.ProjectID = p.ProjectID LEFT OUTER JOIN
                      dbo.General g ON g.OrgID = p.OrgID LEFT OUTER JOIN
                      dbo.ProjectInfo prji ON prji.ProjectInfoID = p.ProjectID
GO

CREATE VIEW [dbo].[vProjects]
AS
SELECT     p.ProjectID, g.Name, g.OrgID, p.Label, g.NameRu
FROM         dbo.Projects AS p INNER JOIN
                      dbo.General AS g ON g.OrgID = p.OrgID
WHERE     (p.isDeleted = 'False') OR
                      (p.isDeleted IS NULL)
